178     public static LineAwesomeRegular findByDescription(String description) {
179         for (LineAwesomeRegular font : values()) {
180             if (font.getDescription().equals(description)) {
181                 return font;
182             }
183         }
184         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
185     }

193     }
