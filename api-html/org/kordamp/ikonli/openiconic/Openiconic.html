251     public static Openiconic findByDescription(String description) {
252         for (Openiconic font : values()) {
253             if (font.getDescription().equals(description)) {
254                 return font;
255             }
256         }
257         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
258     }

260     private String description;
261     private int code;
262 
263     Openiconic(String description, int code) {
264         this.description = description;
265         this.code = code;
266     }
267 
268     @Override
269     public String getDescription() {
270         return description;
271     }
272 
273     @Override
274     public int getCode() {
275         return code;
276     }
