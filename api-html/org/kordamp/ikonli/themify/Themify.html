379     public static Themify findByDescription(String description) {
380         for (Themify font : values()) {
381             if (font.getDescription().equals(description)) {
382                 return font;
383             }
384         }
385         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
386     }
