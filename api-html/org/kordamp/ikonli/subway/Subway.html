333     public static Subway findByDescription(String description) {
334         for (Subway font : values()) {
335             if (font.getDescription().equals(description)) {
336                 return font;
337             }
338         }
339         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
340     }
