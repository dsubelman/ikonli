266     public static LigatureSymbols findByDescription(String description) {
267         for (LigatureSymbols font : values()) {
268             if (font.getDescription().equals(description)) {
269                 return font;
270             }
271         }
272         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
273     }

276     private int code;
277 
278     LigatureSymbols(String description, int code) {
279         this.description = description;
280         this.code = code;
281     }

283     @Override
