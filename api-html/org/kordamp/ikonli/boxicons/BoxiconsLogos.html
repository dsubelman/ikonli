153     public static BoxiconsLogos findByDescription(String description) {
154         for (BoxiconsLogos font : values()) {
155             if (font.getDescription().equals(description)) {
156                 return font;
157             }
158         }
159         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
160     }
