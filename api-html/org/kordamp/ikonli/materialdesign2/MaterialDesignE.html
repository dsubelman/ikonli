198     public static MaterialDesignE findByDescription(String description) {
199         for (MaterialDesignE font : values()) {
200             if (font.getDescription().equals(description)) {
201                 return font;
202             }
203         }
204         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
205     }

208     private int code;

210     MaterialDesignE(String description, String code) {
211         this.description = description;
212         this.code = Integer.parseInt(code, 16);
213     }
