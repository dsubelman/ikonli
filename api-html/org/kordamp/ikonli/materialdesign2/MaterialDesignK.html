086     public static MaterialDesignK findByDescription(String description) {
087         for (MaterialDesignK font : values()) {
088             if (font.getDescription().equals(description)) {
089                 return font;
090             }
091         }
092         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
093     }

095     private String description;
096     private int code;

098     MaterialDesignK(String description, String code) {
099         this.description = description;
100         this.code = Integer.parseInt(code, 16);
101     }
