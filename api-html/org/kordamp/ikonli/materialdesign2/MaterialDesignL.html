209     public static MaterialDesignL findByDescription(String description) {
210         for (MaterialDesignL font : values()) {
211             if (font.getDescription().equals(description)) {
212                 return font;
213             }
214         }
215         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
216     }

218     private String description;
219     private int code;
220 
221     MaterialDesignL(String description, String code) {
222         this.description = description;
223         this.code = Integer.parseInt(code, 16);
224     }
225 
226     @Override
227     public String getDescription() {
228         return description;
229     }
230 
231     @Override
232     public int getCode() {
233         return code;
234     }
