258     public static MaterialDesignH findByDescription(String description) {
259         for (MaterialDesignH font : values()) {
260             if (font.getDescription().equals(description)) {
261                 return font;
262             }
263         }
264         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
265     }

273     }
