161     public static MaterialDesignV findByDescription(String description) {
162         for (MaterialDesignV font : values()) {
163             if (font.getDescription().equals(description)) {
164                 return font;
165             }
166         }
167         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
168     }

170     private String description;
171     private int code;
172 
173     MaterialDesignV(String description, String code) {
174         this.description = description;
175         this.code = Integer.parseInt(code, 16);
176     }

178     @Override

183     @Override
184     public int getCode() {
