135     public static Microns findByDescription(String description) {
136         for (Microns font : values()) {
137             if (font.getDescription().equals(description)) {
138                 return font;
139             }
140         }
141         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
142     }
143 
144     private String description;
145     private int code;
146 
147     Microns(String description, int code) {
148         this.description = description;
149         this.code = code;
150     }
151 
152     @Override
153     public String getDescription() {
154         return description;
155     }
156 
157     @Override
158     public int getCode() {
159         return code;
160     }
