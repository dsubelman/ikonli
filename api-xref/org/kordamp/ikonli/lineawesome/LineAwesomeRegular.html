178     public static LineAwesomeRegular findByDescription(String description) {
179         for (LineAwesomeRegular font : values()) {
180             if (font.getDescription().equals(description)) {
181                 return font;
182             }
183         }
184         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
185     }
186 
187     private String description;
188     private int code;
189 
190     LineAwesomeRegular(String description, int code) {
191         this.description = description;
192         this.code = code;
193     }
194 
195     @Override
196     public String getDescription() {
197         return description;
198     }
199 
200     @Override
201     public int getCode() {
202         return code;
203     }
