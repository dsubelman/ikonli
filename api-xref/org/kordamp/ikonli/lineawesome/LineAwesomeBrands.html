484     public static LineAwesomeBrands findByDescription(String description) {
485         for (LineAwesomeBrands font : values()) {
486             if (font.getDescription().equals(description)) {
487                 return font;
488             }
489         }
490         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
491     }
