856     public static CoreUiBrands findByDescription(String description) {
857         for (CoreUiBrands font : values()) {
858             if (font.getDescription().equals(description)) {
859                 return font;
860             }
861         }
862         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
863     }
