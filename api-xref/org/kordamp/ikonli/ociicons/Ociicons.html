96      Ociicons(String description, int code) {

100 
101     public static Ociicons findByDescription(String description) {
102         for (Ociicons font : values()) {
103             if (font.getDescription().equals(description)) {
104                 return font;
105             }
106         }
107         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
108     }
