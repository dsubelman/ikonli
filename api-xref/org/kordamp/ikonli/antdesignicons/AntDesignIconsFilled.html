245     public static AntDesignIconsFilled findByDescription(String description) {
246         for (AntDesignIconsFilled font : values()) {
247             if (font.getDescription().equals(description)) {
248                 return font;
249             }
250         }
251         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
252     }
253 
254     private String description;
255     private int code;
256 
257     AntDesignIconsFilled(String description, int code) {
258         this.description = description;
259         this.code = code;
260     }
261 
262     @Override
263     public String getDescription() {
264         return description;
265     }
266 
267     @Override
268     public int getCode() {
269         return code;
270     }
