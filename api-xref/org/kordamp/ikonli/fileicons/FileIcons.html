931     public static FileIcons findByDescription(String description) {
932         for (FileIcons font : values()) {
933             if (font.getDescription().equals(description)) {
934                 return font;
935             }
936         }
937         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
938     }
