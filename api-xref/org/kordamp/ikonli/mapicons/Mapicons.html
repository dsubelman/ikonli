202     public static Mapicons findByDescription(String description) {
203         for (Mapicons font : values()) {
204             if (font.getDescription().equals(description)) {
205                 return font;
206             }
207         }
208         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
209     }
210 
211     private String description;
212     private int code;
213 
214     Mapicons(String description, int code) {
215         this.description = description;
216         this.code = code;
217     }
218 
219     @Override
220     public String getDescription() {
221         return description;
222     }
223 
224     @Override
225     public int getCode() {
226         return code;
227     }
