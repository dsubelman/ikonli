344     public static Ionicons4IOS findByDescription(String description) {
345         for (Ionicons4IOS font : values()) {
346             if (font.getDescription().equals(description)) {
347                 return font;
348             }
349         }
350         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
351     }

354     private int code;

357         this.description = description;
358         this.code = code;
359     }
360 
361     @Override
362     public String getDescription() {
363         return description;
364     }
365
