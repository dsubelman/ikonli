171     public static Medicons findByDescription(String description) {
172         for (Medicons font : values()) {
173             if (font.getDescription().equals(description)) {
174                 return font;
175             }
176         }
177         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
178     }
179 
180     private String description;
181     private int code;
182 
183     Medicons(String description, int code) {
184         this.description = description;
185         this.code = code;
186     }
187 
188     @Override
189     public String getDescription() {
190         return description;
191     }
192 
193     @Override
194     public int getCode() {
195         return code;
196     }
