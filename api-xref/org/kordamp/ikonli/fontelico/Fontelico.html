61      public static Fontelico findByDescription(String description) {
62          for (Fontelico font : values()) {
63              if (font.getDescription().equals(description)) {
64                  return font;
65              }
66          }
67          throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
68      }
69  
70      private String description;
71      private int code;
72  
73      Fontelico(String description, int code) {
74          this.description = description;
75          this.code = code;
76      }
77  
78      @Override
79      public String getDescription() {
80          return description;
81      }
82  
83      @Override
84      public int getCode() {
85          return code;
86      }
