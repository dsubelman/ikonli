218     public static Devicons findByDescription(String description) {
219         for (Devicons font : values()) {
220             if (font.getDescription().equals(description)) {
221                 return font;
222             }
223         }
224         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
225     }
226 
227     private String description;
228     private int code;
229 
230     Devicons(String description, int code) {
231         this.description = description;
232         this.code = code;
233     }
234 
235     @Override
236     public String getDescription() {
237         return description;
238     }
239 
240     @Override
241     public int getCode() {
242         return code;
243     }
