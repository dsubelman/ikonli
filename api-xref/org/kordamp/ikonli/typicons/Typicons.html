364     public static Typicons findByDescription(String description) {
365         for (Typicons font : values()) {
366             if (font.getDescription().equals(description)) {
367                 return font;
368             }
369         }
370         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
371     }
372 
373     private String description;
374     private int code;
375 
376     Typicons(String description, int code) {
377         this.description = description;
378         this.code = code;
379     }
380 
381     @Override
382     public String getDescription() {
383         return description;
384     }
385 
386     @Override
387     public int getCode() {
388         return code;
389     }
