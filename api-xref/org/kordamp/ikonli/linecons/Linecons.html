75      private String description;
76      private int code;
77  
78      Linecons(String description, int code) {
79          this.description = description;
80          this.code = code;
81      }
82  
83      public static Linecons findByDescription(String description) {
84          for (Linecons font : values()) {
85              if (font.getDescription().equals(description)) {
86                  return font;
87              }
88          }
89          throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
90      }
91  
92      @Override
93      public String getDescription() {
94          return description;
95      }
96  
97      @Override
98      public int getCode() {
99          return code;
100     }
