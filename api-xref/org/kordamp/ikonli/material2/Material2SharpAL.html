754     public static Material2SharpAL findByDescription(String description) {
755         for (Material2SharpAL font : values()) {
756             if (font.getDescription().equals(description)) {
757                 return font;
758             }
759         }
760         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
761     }
