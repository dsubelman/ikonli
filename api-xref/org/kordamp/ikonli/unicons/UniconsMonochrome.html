311     public static UniconsMonochrome findByDescription(String description) {
312         for (UniconsMonochrome font : values()) {
313             if (font.getDescription().equals(description)) {
314                 return font;
315             }
316         }
317         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
318     }

321     private int code;

324         this.description = description;
325         this.code = code;
326     }
327 
328     @Override
329     public String getDescription() {
330         return description;
331     }
332 
333     @Override
334     public int getCode() {
335         return code;
336     }
