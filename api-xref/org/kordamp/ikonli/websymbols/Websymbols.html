193     public static Websymbols findByDescription(String description) {
194         for (Websymbols font : values()) {
195             if (font.getDescription().equals(description)) {
196                 return font;
197             }
198         }
199         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
200     }
201 
202     private String description;
203     private int code;
204 
205     Websymbols(String description, int code) {
206         this.description = description;
207         this.code = code;
208     }
209 
210     @Override
211     public String getDescription() {
212         return description;
213     }
214 
215     @Override
216     public int getCode() {
217         return code;
218     }
