275     public static WeatherIcons findByDescription(String description) {
276         for (WeatherIcons font : values()) {
277             if (font.getDescription().equals(description)) {
278                 return font;
279             }
280         }
281         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
282     }

285     private int code;

288         this.description = description;
289         this.code = code;
290     }
