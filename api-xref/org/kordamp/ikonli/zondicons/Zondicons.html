325     public static Zondicons findByDescription(String description) {
326         for (Zondicons font : values()) {
327             if (font.getDescription().equals(description)) {
328                 return font;
329             }
330         }
331         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
332     }
333 
334     private String description;
335     private int code;
336 
337     Zondicons(String description, int code) {
338         this.description = description;
339         this.code = code;
340     }
341 
342     @Override
343     public String getDescription() {
344         return description;
345     }
346 
347     @Override
348     public int getCode() {
349         return code;
350     }
