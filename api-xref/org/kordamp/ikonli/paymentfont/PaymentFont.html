144     public static PaymentFont findByDescription(String description) {
145         for (PaymentFont font : values()) {
146             if (font.getDescription().equals(description)) {
147                 return font;
148             }
149         }
150         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
151     }
152 
153     private String description;
154     private int code;
155 
156     PaymentFont(String description, int code) {
157         this.description = description;
158         this.code = code;
159     }
160 
161     @Override
162     public String getDescription() {
163         return description;
164     }
165 
166     @Override
167     public int getCode() {
168         return code;
169     }
