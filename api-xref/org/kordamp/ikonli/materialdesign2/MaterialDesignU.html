66      public static MaterialDesignU findByDescription(String description) {
67          for (MaterialDesignU font : values()) {
68              if (font.getDescription().equals(description)) {
69                  return font;
70              }
71          }
72          throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
73      }
74  
75      private String description;
76      private int code;
77  
78      MaterialDesignU(String description, String code) {
79          this.description = description;
80          this.code = Integer.parseInt(code, 16);
81      }
82  
83      @Override
84      public String getDescription() {
85          return description;
86      }
87  
88      @Override
89      public int getCode() {
90          return code;
91      }
