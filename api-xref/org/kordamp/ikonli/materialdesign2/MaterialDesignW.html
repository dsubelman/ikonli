208     public static MaterialDesignW findByDescription(String description) {
209         for (MaterialDesignW font : values()) {
210             if (font.getDescription().equals(description)) {
211                 return font;
212             }
213         }
214         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
215     }
216 
217     private String description;
218     private int code;
219 
220     MaterialDesignW(String description, String code) {
221         this.description = description;
222         this.code = Integer.parseInt(code, 16);
223     }
224 
225     @Override
226     public String getDescription() {
227         return description;
228     }
229 
230     @Override
231     public int getCode() {
232         return code;
233     }
