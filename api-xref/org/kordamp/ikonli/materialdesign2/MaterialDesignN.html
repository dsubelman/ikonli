197     public static MaterialDesignN findByDescription(String description) {
198         for (MaterialDesignN font : values()) {
199             if (font.getDescription().equals(description)) {
200                 return font;
201             }
202         }
203         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
204     }
205 
206     private String description;
207     private int code;
208 
209     MaterialDesignN(String description, String code) {
210         this.description = description;
211         this.code = Integer.parseInt(code, 16);
212     }
213 
214     @Override
215     public String getDescription() {
216         return description;
217     }
218 
219     @Override
220     public int getCode() {
221         return code;
