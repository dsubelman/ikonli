190     public static MaterialDesignG findByDescription(String description) {
191         for (MaterialDesignG font : values()) {
192             if (font.getDescription().equals(description)) {
193                 return font;
194             }
195         }
196         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
197     }
198 
199     private String description;
200     private int code;
201 
202     MaterialDesignG(String description, String code) {
203         this.description = description;
204         this.code = Integer.parseInt(code, 16);
205     }
206 
207     @Override
208     public String getDescription() {
209         return description;
210     }
211 
212     @Override
213     public int getCode() {
214         return code;
215     }
