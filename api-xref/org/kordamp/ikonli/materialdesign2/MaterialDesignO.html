70      public static MaterialDesignO findByDescription(String description) {
71          for (MaterialDesignO font : values()) {
72              if (font.getDescription().equals(description)) {
73                  return font;
74              }
75          }
76          throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
77      }
78  
79      private String description;
80      private int code;
81  
82      MaterialDesignO(String description, String code) {
83          this.description = description;
84          this.code = Integer.parseInt(code, 16);
85      }
86  
87      @Override
88      public String getDescription() {
89          return description;
90      }
91  
92      @Override
93      public int getCode() {
94          return code;
95      }
