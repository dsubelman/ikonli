545     public static HawconsStroke findByDescription(String description) {
546         for (HawconsStroke font : values()) {
547             if (font.getDescription().equals(description)) {
548                 return font;
549             }
550         }
551         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
552     }
